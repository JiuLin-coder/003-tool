git config --global user.name "JiuLin"  //配置用户信息
git config --global user.email "XX@XX"

git init                                //初始化,创建仓库

git log                                 //查看提交信息
git log --oneline                       //查看简洁版提交历史
git log --all --graph --decorate --oneline   //更好看点的查看 
git status                  		// 查看当前状态

git add .                               //暂存所有改动
git add <filename>                      //暂存指定文件
git commit -m "描述"                    //提交，保存为一个版本，描述要认真写，

git reset --hard <log时，commit后面的那串ID>	//在该线 回退到某一commit版本
 
# 有版本就可以创建出不同的链接
git branch <filename>				//新建一个分支	，在复制当前文件并创建一个新分支,此时好像只有一个新箭头，而没有commit。merge后分支仍保留
git branch -a						//查看所有分支
git branch -b <filename>			//创建并进入filename分支
git log --oneline                      		 //查看简洁版提交历史
git checkout <filename>			//切换到filename分支里
git merge <filename>            		// 把某分支合并到当前分支    若两分支修改同一个代码则会报错
git branch -d <filename>      //删掉分支

#推送到自己的网盘里
git add .
git commit -m "描述"
git branch -M main				//把主分支修改成main ，并切换到main，        //三步
git remote add origin <url>              	//相当于给你的本地一个github的地址
git push -u origin <branchName>:<remote's branchName>	//推送到github里     	 ，把你本地代码的main分支给推送上去

#参与开源项目      //两个库必须完全（内容，log）相同才能push，所以只能clone
# 复制代码到本地
  # 1.先去github 点fork 把项目代码复制到自己仓库里，类似branch
git clone <url> 					  //克隆远程仓库到本地，并且remote 的origin就会变成你的仓库的
  # 并且自动把remote 的origin改为自己仓库的链接
git remote -v                                          //查看已有的远程仓库链接
  # 复制项目的链接
git remote add upstream <url>		//url为仓库的链接， 其实upstream也可以是别的，但是下面push就要跟它对应
  # 修改，然后add, commit, branch   
git push -u upstream <branchName>:<remote's branchName>    //把新分支的项目推送给远程服务器


#若是你的仓库并收到新branch（git push），虽然有新的，但是此时只是分支，还没有合并到远程仓库的main分支
点pull requests ,点击新分支,比较若绿色对号，则create pr

#确保可以pr则要有最新的远程代码版本 到本地
#（其实就是远程仓库合并新分支，本地也是同理，即有最新，就把最新的分支推送并合并到本地）
git fetch upstream （查看是否有新分支，并获取新分支信息，但是并没有合并，）
git merge upstream/main 

git pull  //上面两条效果合一
 #再执行pr

git pull origin main        # 拉取远端最新代码
